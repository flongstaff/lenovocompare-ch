**workers/src/index.ts:**

Hono app with D1 binding, CORS middleware, and GET /api/prices endpoint.

```typescript
import { Hono } from "hono";
import { cors } from "hono/cors";

type Bindings = {
  DB: D1Database;
};

interface PriceRow {
  id: string;
  laptop_id: string;
  retailer: string;
  price_chf: number;
  price_type: string | null;
  url: string | null;
  note: string | null;
  date_added: string;
  is_user_added: number;
  source: string;
}

const app = new Hono<{ Bindings: Bindings }>();

// CORS: allow GitHub Pages origin + localhost dev
// Per CONTEXT.md: Workers and static site are separate origins -- CORS headers required
app.use(
  "*",
  cors({
    origin: ["https://flongstaff.github.io", "http://localhost:3000"],
    allowMethods: ["GET", "OPTIONS"],
    allowHeaders: ["Content-Type"],
    maxAge: 86400,
  }),
);

// Health check
app.get("/", (c) => c.json({ status: "ok", service: "lenovocompare-prices" }));

// GET /api/prices -- return all current prices
app.get("/api/prices", async (c) => {
  try {
    const { results } = await c.env.DB.prepare("SELECT * FROM prices ORDER BY date_added DESC").all<PriceRow>();

    // Transform snake_case DB rows to camelCase matching SwissPrice interface
    const prices = (results ?? []).map((row) => ({
      id: row.id,
      laptopId: row.laptop_id,
      retailer: row.retailer,
      price: row.price_chf,
      url: row.url,
      dateAdded: row.date_added,
      isUserAdded: row.is_user_added === 1,
      priceType: row.price_type,
      note: row.note,
    }));

    return c.json(prices);
  } catch (err) {
    console.error("[GET /api/prices] Error:", err);
    return c.json({ error: "Failed to fetch prices" }, 500);
  }
});

// GET /api/prices/:laptopId -- return prices for a specific model
app.get("/api/prices/:laptopId", async (c) => {
  const laptopId = c.req.param("laptopId");
  try {
    const { results } = await c.env.DB.prepare("SELECT * FROM prices WHERE laptop_id = ? ORDER BY date_added DESC")
      .bind(laptopId)
      .all<PriceRow>();

    const prices = (results ?? []).map((row) => ({
      id: row.id,
      laptopId: row.laptop_id,
      retailer: row.retailer,
      price: row.price_chf,
      url: row.url,
      dateAdded: row.date_added,
      isUserAdded: row.is_user_added === 1,
      priceType: row.price_type,
      note: row.note,
    }));

    return c.json(prices);
  } catch (err) {
    console.error(`[GET /api/prices/${laptopId}] Error:`, err);
    return c.json({ error: "Failed to fetch prices" }, 500);
  }
});

export default app;
```

The API transforms D1's snake_case columns to camelCase matching the existing `SwissPrice` interface shape, so the frontend hook can consume the response directly.

  <action>
Read `data/seed-prices.ts` to extract all ~205 seed price entries. Generate a SQL file `workers/seeds/seed-prices.sql` with INSERT statements for each entry.

The SQL must translate from the TypeScript SwissPrice interface to the D1 schema:

- `id` -> `id` (TEXT)
- `laptopId` -> `laptop_id` (TEXT)
- `retailer` -> `retailer` (TEXT)
- `price` -> `price_chf` (REAL)
- `priceType` -> `price_type` (TEXT, nullable)
- `url` -> `url` (TEXT, nullable)
- `note` -> `note` (TEXT, nullable)
- `dateAdded` -> `date_added` (TEXT)
- `isUserAdded` -> `is_user_added` (INTEGER, 0 or 1)
- `source` -> hardcode as `'seed'` for all entries

Use INSERT OR IGNORE to make the seed idempotent (safe to re-run).

Escape single quotes in note/url fields by doubling them (`'` -> `''`).

Generate the SQL by reading the TypeScript source file and constructing INSERT statements. Do NOT hand-write the SQL -- parse the actual data to ensure laptop IDs match.

Format:

```sql
-- Seed prices generated from data/seed-prices.ts
-- Run: wrangler d1 execute lenovocompare-prices --file seeds/seed-prices.sql --local

INSERT OR IGNORE INTO prices (id, laptop_id, retailer, price_chf, price_type, url, note, date_added, is_user_added, source)
VALUES ('sp-1', 'x1-carbon-gen12', 'Digitec', 1799.0, 'retail', 'https://...', '...', '2024-11-15', 0, 'seed');
-- ... one INSERT per price entry
```
